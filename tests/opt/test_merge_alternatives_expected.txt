# Test merge alternatives optimization

/abc|bc|x|ab.|bc./
    merge_alternatives, print
    /ab(?:c|.)|bc(?:|.)|x/

/x(abc|bc|x|ab.|bc.)|xy/
    merge_alternatives, print
    /x(?:(ab(?:c|.)|bc(?:|.)|x)|y)/

/abcdefgh|abcdefghi/
    merge_alternatives, print
    /abcdefgh(?:|i)/

/abcdefgh|.|abcdefghi/
    merge_alternatives, print
    /abcdefgh|.|abcdefghi/

/abcdefgh|.|abcdefgh|abcdefghi/
    merge_alternatives, print
    /abcdefgh|.|abcdefgh(?:|i)/

/abc|(?i)A(?-i)|abcd/
    merge_alternatives, print
    /abc|(?i)A|(?-i)abcd/

/(ab(cd|c)|ab(cd|c))/
    merge_alternatives, print
    /(ab(?:(c(?:d|))|(c(?:d|))))/

/(?=abc|ab|abd)/
    merge_alternatives, print
    /(?=ab(?:c||d))/

/(?<name>abc|ab.|abc)/
    merge_alternatives, print
    /(?<name>ab(?:c|.|c))/
