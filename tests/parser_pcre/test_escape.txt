# \c - control-x
# \g - unified backreference
# \k - named backreference
# \o - octal number enclosed in braces
# \p{...} - unicode property name
# \x - hexadecimal number

    print

# \P{...} - negated unicode property name
# \Q - part of \Q...\E sequence
# \E - stray \E is removed
